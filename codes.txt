    <script>
      // ====== CUSTOM SETTINGS ======
      // ðŸŽšï¸ Adjust the intensity of the 3D movement
      const movementStrength = 10; // Try between 5â€“25 for more or less motion

      // Select the main group
      const tiltGroup = document.getElementById("tiltGroup");

      // Smooth 3D tilt based on cursor movement
      document.addEventListener("mousemove", (e) => {
        const { innerWidth, innerHeight } = window;
        const x = (e.clientX / innerWidth - 0.5) * 2; // range -1 to 1
        const y = (e.clientY / innerHeight - 0.5) * 2; // range -1 to 1

        const rotateX = y * movementStrength * -1; // reverse for natural tilt
        const rotateY = x * movementStrength;

        tiltGroup.style.transform = `rotateX(${rotateX}deg) rotateY(${rotateY}deg)`;
      });

      // Reset when mouse leaves
      document.addEventListener("mouseleave", () => {
        tiltGroup.style.transform = "rotateX(0deg) rotateY(0deg)";
      });
    </script>
